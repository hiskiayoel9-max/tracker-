import com.google.android.gms.location.* import android.os.Looper class MainActivity : AppCompatActivity() { private lateinit var fusedLocationClient: FusedLocationProviderClient private lateinit var locationCallback: LocationCallback override fun onCreate(savedInstanceState: Bundle?) { super.onCreate(savedInstanceState) fusedLocationClient = LocationServices.getFusedLocationProviderClient(this) startTracking() } private fun startTracking() { // Pengaturan untuk akurasi maksimal val locationRequest = LocationRequest.Builder(Priority.PRIORITY_HIGH_ACCURACY, 5000) // update tiap 5 detik .setWaitForAccurateLocation(true) // Menunggu hingga sinyal benar-benar stabil .setMinUpdateIntervalMillis(2000) .build() locationCallback = object : LocationCallback() { override fun onLocationResult(locationResult: LocationResult) { for (location in locationResult.locations) { // Di sini Anda mendapatkan koordinat (Lat, Long) println("Lokasi Akurat: ${location.latitude}, ${location.longitude}") println("Margin Error: ${location.accuracy} meter") } } } // Jalankan pelacakan fusedLocationClient.requestLocationUpdates( locationRequest, locationCallback, Looper.getMainLooper() ) } }
